k8s/06-loader-job.yaml

```
apiVersion: batch/v1
kind: Job
metadata:
  name: data-loader
spec:
  template:
    spec:
      restartPolicy: Never
      containers:
      - name: loader
        image: data-loader:latest
        env:
        - name: DB_HOST
          value: "mysql-service"
        - name: DB_PORT
          value: "3306"
        - name: DB_USER
          value: "root"
        - name: DB_PASSWORD
          valueFrom:
            secretKeyRef:
              name: mysql-secret
              key: root-password
        - name: DB_NAME
          valueFrom:
            configMapKeyRef:
              name: mysql-config
              key: database
      # Добавляем задержку, чтобы MySQL успел запуститься
      initContainers:
      - name: wait-for-mysql
        image: mysql:8.0
        command: ['sh', '-c', 'until mysql -h mysql-service -u root -p$DB_PASSWORD -e "SELECT 1"; do echo waiting for mysql; sleep 2; done']
        env:
        - name: DB_PASSWORD
          valueFrom:
            secretKeyRef:
              name: mysql-secret
              key: root-password
  backoffLimit: 5
  ```
